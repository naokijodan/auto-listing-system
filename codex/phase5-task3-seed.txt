async function seedItemSpecificsFields() {
  console.log('Seeding ItemSpecificsField...');
  let count = 0;
  for (const field of INITIAL_DATA) {
    await prisma.itemSpecificsField.upsert({
      where: {
        category_fieldName: {
          category: field.category,
          fieldName: field.fieldName,
        },
      },
      create: {
        category: field.category,
        tagJp: field.tagJp,
        fieldName: field.fieldName,
        fieldType: field.fieldType,
        priority: field.priority,
        notes: field.notes || null,
      },
      update: {
        tagJp: field.tagJp,
        fieldType: field.fieldType,
        priority: field.priority,
        notes: field.notes || null,
      },
    });
    count++;
  }
  console.log(`  ItemSpecificsField: ${count} records upserted`);
}

async function seedBrands() {
  console.log('Seeding Brand...');
  let count = 0;
  for (const brand of BRAND_DATA) {
    await prisma.brand.upsert({
      where: { name: brand.name },
      create: {
        name: brand.name,
        jpNames: brand.jpNames,
        country: brand.country || null,
        parentBrand: brand.parentBrand || null,
        categories: brand.categories || [],
        isMaterial: brand.isMaterial || false,
      },
      update: {
        jpNames: brand.jpNames,
        country: brand.country || null,
        parentBrand: brand.parentBrand || null,
        categories: brand.categories || [],
        isMaterial: brand.isMaterial || false,
      },
    });
    count++;
  }
  console.log(`  Brand: ${count} records upserted`);
}

async function main() {
  console.log('Starting Item Specifics seed...');
  await seedItemSpecificsFields();
  await seedBrands();
  console.log('Seed complete!');
}

main()
  .catch((e) => {
    console.error('Seed failed:', e);
    process.exit(1);
  })
  .finally(async () => {
    await prisma.$disconnect();
  });

## 重要な指示
1. BRAND_DATAの中身は空の配列になっている。Config_IS.gsの246行目〜786行目のIS_BRAND_DICTの内容を全て読み取り、TypeScriptの配列としてBRAND_DATAに入れること。
2. GASの `{name: 'Rolex', jp_names: [...], country: '...'}` を TypeScriptの `{name: 'Rolex', jpNames: [...], country: '...'}` に変換（jp_names → jpNames, parent_brand → parentBrand, is_material → isMaterial, category → categories）
3. 約540件あるので漏れなく全て入れること
4. 実行方法: `npx tsx packages/database/prisma/seed-item-specifics.ts`（ただし本番実行はユーザーが行う）
